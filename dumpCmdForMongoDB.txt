PART 1 — ATLAS → LOCAL (DUMP FROM CLOUD)

Dump ALL databases
mongodump --uri="mongodb+srv://user:pass@cluster.mongodb.net" --out=backup

Dump ONE database
mongodump --uri="MONGODB_URL=mongodb+srv://faraz_32729:farazMongodb@cluster0.mrwq60z.mongodb.net/luckyOneMall" --out=backup

Dump ONE collection
mongodump \
  --uri="mongodb+srv://user:pass@cluster.mongodb.net/your-db-name" \
  --collection=your-collection-name \
  --out=backup

PART 2 — RESTORE TO LOCAL MONGODB

Restore ALL databases
mongorestore --uri="mongodb://127.0.0.1:27017" backup/

Restore ONE database
mongorestore --uri="mongodb://127.0.0.1:27017" backup/your-db-name

Restore ONE collection
mongorestore \
  --uri="mongodb://127.0.0.1:27017" \
  --nsInclude="your-db-name.your-collection-name" \
  backup/



PART 3 — LOCAL → ATLAS (SEND BACK)

Upload ALL databases
mongorestore --uri="mongodb+srv://user:pass@cluster.mongodb.net" backup/

Upload ONE database
mongorestore \
  --uri="mongodb+srv://user:pass@cluster.mongodb.net" \
  backup/your-db-name

Upload ONE collection
mongorestore \
  --uri="mongodb+srv://user:pass@cluster.mongodb.net" \
  --nsInclude="your-db-name.your-collection-name" \
  backup/
